018package org.apache.commons.codec;
019
020/**
021 * Provides the highest level of abstraction for Decoders.
022 * <p>
023 * This is the sister interface of {@link Encoder}. All Decoders implement this common generic interface.
024 * Allows a user to pass a generic Object to any Decoder implementation in the codec package.
025 * <p>
026 * One of the two interfaces at the center of the codec package.
027 *
028 */
029public interface Decoder {
030
031    /**
032     * Decodes an "encoded" Object and returns a "decoded" Object. Note that the implementation of this interface will
033     * try to cast the Object parameter to the specific type expected by a particular Decoder implementation. If a
034     * {@link ClassCastException} occurs this decode method will throw a DecoderException.
035     *
036     * @param source
037     *            the object to decode
038     * @return a 'decoded" object
039     * @throws DecoderException
040     *             a decoder exception can be thrown for any number of reasons. Some good candidates are that the
041     *             parameter passed to this method is null, a param cannot be cast to the appropriate type for a
042     *             specific encoder.
043     */
044    Object decode(Object source) throws DecoderException;
045}
